const SHEET_NAME = 'List1'; function doGet(e) { const action = e.parameter.action; let result; try { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); if (action === 'get') { const data = sheet.getDataRange().getValues(); const rows = data.slice(1).map(r => ({ datum: r[0] ? Utilities.formatDate(new Date(r[0]), 'Europe/Prague', 'yyyy-MM-dd') : '', vyrobek: r[1] || '', objem: r[2] || '', vyroba: r[3] ? Utilities.formatDate(new Date(r[3]), 'Europe/Prague', 'yyyy-MM-dd') : '', sarze: r[4] || '', vzhled: r[5] || '', vune: r[6] || '', chut: r[7] || '', poznamka: r[8] || '', hodnotitele: r[9] ? r[9].toString().split('+') : [] })).filter(r => r.datum); result = { ok: true, data: rows }; } else if (action === 'add') { const d = JSON.parse(e.parameter.payload); sheet.appendRow([ d.datum, d.vyrobek, d.objem, d.vyroba, d.sarze, d.vzhled, d.vune, d.chut, d.poznamka || '', d.hodnotitele.join('+'), d.zadal || '' ]); result = { ok: true }; } else { result = { ok: false, error: 'Neznámá akce' }; } } catch (err) { result = { ok: false, error: err.toString() }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); }
